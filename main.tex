\begin{center}
    {\Huge \textbf{Einstein's Energy-Momentum Relation}}
    \vspace{1.5cm}
    
    {\LARGE $ p^\mu p_\mu = m^2 c^2 $}
    
    \vspace{1cm}
    
    {\Large Equivalent form:}
    
    \vspace{0.5cm}
    
    {\LARGE $ E^2 = p^2 c^2 + m^2 c^4 $}
    
    \vspace{1.5cm}
    
    Where:
    \begin{itemize}
        \item $p^\mu$ = Four-momentum
        \item $m$ = Rest mass
        \item $c$ = Speed of light
        \item $E$ = Energy
        \item $p$ = Magnitude of three-momentum
    \end{itemize}
    
\end{center}
